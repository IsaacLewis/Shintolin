  def inhabitant_ids
    query = "SELECT `accounts`.`id` " +
      "FROM `users` , `accounts` " +
      "WHERE `users`.`id` = `accounts`.`id` " +
      "AND `accounts`.`settlement_id` = '#{mysql_id}' " +
      "AND `users`.`active` = '1'"
    result = $mysql.query(query)
    inhabitants = []
    result.each_hash {|row| inhabitants << row['id']}
    inhabitants
  end
